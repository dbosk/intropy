\section{Introduktion}

\begin{frame}
  \begin{block}{MATrix LABoratory, MATLAB}%
    \foreignblockquote{english}[Wikipedia, MATLAB]{MATLAB (an abbreviation of 
    "matrix laboratory") is a proprietary multi-paradigm programming language 
  and numerical computing environment developed by MathWorks.}
  \end{block}
\end{frame}

\begin{frame}
  \begin{remark}[Proprietärt]
    \begin{itemize}
      \item Åtkomst till Matlab Online via KTH: 
        \url{https://matlab.mathworks.com}.

      \item GNU Octave är öppen källkod.
      \item Dock matchar inte allt perfekt.
    \end{itemize}
  \end{remark}
\end{frame}

\begin{frame}
  \begin{block}{Variabler}
    \begin{itemize}
      \item Alla variabler är \(n\times m\)-matriser.
    \end{itemize}
    \begin{description}
      \item[Matris] \(n > 1, m > 1\)
      \item[Vektor] \(n > 1, m = 1\) eller \(n = 1, m > 1\)
      \item[Skalär] \(n = m = 1\)
    \end{description}
  \end{block}
\end{frame}

\begin{frame}
  \begin{example}[Vektor]
    \begin{columns}
      \begin{column}{0.4\columnwidth}
        \mintinline{matlab}{vector = [1, 2, 3, 4]}
      \end{column}
      \begin{column}{0.4\columnwidth}
        $$\begin{bmatrix}
          1 & 2 & 3 & 4
        \end{bmatrix}$$
      \end{column}
    \end{columns}
    \begin{columns}
      \begin{column}{0.4\columnwidth}
        \mintinline{matlab}{vectorT = [1; 2; 3; 4]}
      \end{column}
      \begin{column}{0.4\columnwidth}
        $$\begin{bmatrix}
          1 \\ 2 \\ 3 \\ 4
        \end{bmatrix}$$
      \end{column}
    \end{columns}
  \end{example}
\end{frame}

\begin{frame}
  \begin{example}[Matris]
    \begin{columns}
      \begin{column}{0.4\columnwidth}
        \mintinline{matlab}{matrix = [1, 2; 3, 4]}
      \end{column}
      \begin{column}{0.4\columnwidth}
        $$\begin{bmatrix}
          1 & 2 \\
          3 & 4
        \end{bmatrix}$$
      \end{column}
    \end{columns}
  \end{example}
\end{frame}

\begin{frame}
  \begin{example}[Skalär]
    \mintinline{matlab}{scalar = 2} eller
    \mintinline{matlab}{scalar = [2]}
  \end{example}
\end{frame}

\begin{frame}
  \begin{remark}[Egenheter]
    \begin{itemize}
      \item Matlabfiler har ändelsen \enquote{.m}.
      \item Alla rader måste ha \enquote{;}, annars skrivs resultatet ut på 
        skärmen.
      \item Matlab beter sig inte riktigt som Python alla gånger.
      \item Mycker görs med globala variabler.
      \item Använd \mintinline{matlab}{clean} för att rensa variabler.
    \end{itemize}
  \end{remark}
\end{frame}


\section{Operationer}

\begin{frame}[fragile]
  \begin{example}[Matrisoperationer]
    \begin{columns}
      \begin{column}{0.4\columnwidth}
        \begin{minted}{matlab}
          A = [1, 2]
          B = [2, 3]
          C = A * B
        \end{minted}
      \end{column}
      \begin{column}{0.4\columnwidth}
        \( C =
        \begin{bmatrix}1 & 2\end{bmatrix}\times
        \begin{bmatrix}2 & 3\end{bmatrix} = ?
        \)
      \end{column}
    \end{columns}
    \begin{columns}
      \begin{column}{0.4\columnwidth}
        \begin{minted}{matlab}
          A = [1, 2]
          B = [2, 3]
          C = A * B'
        \end{minted}
      \end{column}
      \begin{column}{0.4\columnwidth}
        \( C =
        \begin{bmatrix}1 & 2\end{bmatrix}\times
        \begin{bmatrix}2 \\ 3\end{bmatrix} = 8
        \)
      \end{column}
    \end{columns}
    \begin{columns}
      \begin{column}{0.4\columnwidth}
        \begin{minted}{matlab}
          A = [1, 2]
          B = [2, 3]
          C = A .* B
        \end{minted}
      \end{column}
      \begin{column}{0.4\columnwidth}
        \( C =
        \begin{bmatrix}2 & 6\end{bmatrix}
        \)
      \end{column}
    \end{columns}
  \end{example}
\end{frame}

\begin{frame}[fragile]
  \begin{example}[Motsvarande Pythons range]
    \begin{columns}
      \begin{column}{0.4\columnwidth}
        \begin{minted}{matlab}
          A = 1:2
          B = 1:2:6
        \end{minted}
      \end{column}
      \begin{column}{0.4\columnwidth}
        \begin{align*}
          A &= \begin{bmatrix}1 & 2\end{bmatrix} \\
          B &= \begin{bmatrix}1 & 3 & 5\end{bmatrix}
        \end{align*}
      \end{column}
    \end{columns}
  \end{example}
\end{frame}

\section{In-, utmatning}

\begin{frame}[fragile]
  \begin{example}[Utmatning]
    \begin{minted}{matlab}
      disp("Hej och välkommen!")
      A = [1 2];
      disp(A)
    \end{minted}
  \end{example}
\end{frame}

\begin{frame}[fragile]
  \begin{example}[Strängar]
    \begin{minted}{matlab}
      A = [1 2];
      B = [2 3];
      C = A*B';
      msg = strcat(num2str(A), " + ", num2str(B), " = ", num2str(C));
      disp(msg)
    \end{minted}
  \end{example}
\end{frame}

\begin{frame}[fragile]
  \begin{example}[Strängar]
    \begin{minted}{matlab}
      A = [1 2];
      B = input("B = ");
      C = A*B;
      msg = strcat(num2str(A), " + ", num2str(B), " = ", num2str(C));
      disp(msg)
    \end{minted}
  \end{example}
\end{frame}
