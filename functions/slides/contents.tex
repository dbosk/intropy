\section{Funktioner}

\subsection{Vad är bra med funktioner?}

\begin{frame}[fragile]
  \begin{block}{Funktioner}
    \begin{itemize}
      \item Som \enquote{miniprogram} som går att återanvända.
      \item Gör att vi kan minimera kodupprepningar.
      \item Ger färre problem, underlättar underhåll och utbyggnad.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}[fragile]
  \begin{example}[Summera]
    \begin{lstlisting}
l = [1, 2, 3, 4]
print(sum(l))
    \end{lstlisting}
  \end{example}

  \begin{example}[Konkatenera]
    \begin{lstlisting}
l = ["flag", "pole", "polishing"]
print(sum(l))
    \end{lstlisting}
  \end{example}
\end{frame}

\begin{frame}[fragile]
  \begin{example}[Summera själv]
    \begin{lstlisting}
summa = 0

for i in lst:
  summa += i
    \end{lstlisting}
  \end{example}
\end{frame}
