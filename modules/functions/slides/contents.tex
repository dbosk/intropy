\section{Funktioner}

\subsection{Vad är bra med funktioner?}

\begin{frame}[fragile]
  \begin{block}{Funktioner}
    \begin{itemize}
      \item Som \enquote{miniprogram} som går att återanvända.
      \item Gör att vi kan minimera kodupprepningar.
      \item Ger färre problem, underlättar underhåll och utbyggnad.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}[fragile]
  \begin{example}[Summera]
    \begin{minted}{python}
l = [1, 2, 3, 4]
print(sum(l))
    \end{minted}
  \end{example}

  \begin{example}[Konkatenera]
    \begin{minted}{python}
l = ["flag", "pole", "polishing"]
print(sum(l))
    \end{minted}
  \end{example}
\end{frame}

\begin{frame}[fragile]
  \begin{example}[Summera själv]
    \begin{minted}{python}
summa = 0

for i in lst:
  summa += i
    \end{minted}
  \end{example}
\end{frame}
